20void Cuerpo::AdicioneFuerzaGravitacional(Cuerpo &otroCuerpo) {
21 vector3D dr = otroCuerpo.r - r; // Vector de r a otroCuerpo.r
22 double distancia_cubed = std::pow(dr.norm(), 3);
23
24 // Evitar división por cero si los cuerpos están en la misma posición (aunque verificarDatos debería prevenir esto)
25 if (distancia_cubed < 1e-12) { // Un umbral pequeño para evitar problemas numéricos
26 return;
27 }
28
29 // Fuerza F = G * m1 * m2 * (r2 - r1) / |r2 - r1|^3
30 vector3D F_grav = dr * (G * m * otroCuerpo.m / distancia_cubed);
31 F += F_grav;
32}
