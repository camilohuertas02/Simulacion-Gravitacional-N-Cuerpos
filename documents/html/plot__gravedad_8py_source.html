50def get_num_bodies_from_header(filepath):
51 """
52 Lee la primera línea de un archivo de datos para determinar el número de cuerpos.
53 """
54 try:
55 with open(filepath, 'r') as f:
56 header = f.readline()
57 num_cols = len(header.lstrip('#').split())
58 num_bodies = (num_cols - 4) // 4
59 return num_bodies
60 except Exception as e:
61 print(f"Error al leer la cabecera del archivo para determinar N: {e}")
62 return None
